is.Define.module('RozvrhKalendar', function(ns, $) {
	var _data = undefined; // data ziskana z prvniho volani init() ulozime sem a pouzijeme je pri vyvolani initu loaderem
	var handlery_navazane = false;
	var skryte_typy_udalosti = [];
	var posledni_typ_zobrazeni;
	var pocet_requestu_zakazujicich_typy = 0;

	function otevri_sync_token_modal() {
		try {
			$('#sync_token_modal').foundation('open');
		} catch (e) {
			is.Rozvrh.rethrow_exception(e, 'foundation(open) na modalnim okne se sync tokenem');
		}

		$('#sync_token_modal').find('input')[0].select();
	}

	function skryj_udalosti(typy) {
		if (typy) {
			skryte_typy_udalosti = skryte_typy_udalosti.concat(typy.filter(function(t) {
				return skryte_typy_udalosti.indexOf(t) === -1;
			}));
			skryte_typy_udalosti.sort();
		}

		$('#kalendar').fullCalendar('removeEvents', function(o) {
			return (typy || skryte_typy_udalosti).some(function(t) {
				return o.className.indexOf('ev_' + t) > -1;
			});
		});
	}

	function zobraz_udalosti(typy) {
		if (typy) {
			typy.forEach(function(t) {
				var ix = skryte_typy_udalosti.indexOf(t);

				if (ix > -1 && $('.legenda div.toggle-switch.sw_' + t).length > 0) {
					skryte_typy_udalosti.splice(ix, 1);
				}
			});
		}

		try {
			$('#kalendar').fullCalendar('removeEvents'); // pred renderovanim nutno vsechno skryt
		} catch (e) {
			is.Rozvrh.rethrow_exception(e, 'fullCalendar(removeEvents)');
		}
		try {
			$('#kalendar').fullCalendar('renderEvents', _data);
		} catch (e) {
			is.Rozvrh.rethrow_exception(e, 'fullCalendar(renderEvents)');
		}

		if (skryte_typy_udalosti.length > 0) {
			skryj_udalosti();
		}
	}

	return {
		init: function(data, skryte_typy_pref, typ_zobrazeni_pref) {
			_data = data || _data;

			if (typeof _data === 'undefined') {
				return;
			}

			if (typeof skryte_typy_pref !== 'undefined') {
				skryte_typy_udalosti = skryte_typy_pref;
			}
			var naplanovano_skryti_udalosti = skryte_typy_udalosti.length > 0;

			if (typeof typ_zobrazeni_pref !== 'undefined') {
				posledni_typ_zobrazeni = typ_zobrazeni_pref;
			}

			// spocteme min. a max. datum v _data, neumoznime se dostat nikam jinam
			// pro zobrazeni Den omezime radky brzo rano a pozde vecer
			// defaultni hodnoty nastavuji na 9 a 19, ale za cyklem (ktery tyto hodnoty)
			// pripadne zmeni, se interval jeste posouva o hodinu na obe strany
			// tj. je-li kalendar prazdny, zobrazuje se 8:00 - 20:00
			var validRange = {},
				minTime = 9,
				maxTime = 19;

			// pouze slovni oznaceni pro indexy v datum.split(...), at je kod citelnejsi
			var d = {
				Rok: 0,
				Mes: 1,
				Den: 2,
				Hod: 3,
				Min: 4,
				Sek: 5
			};

			for (var i = 0, o; o = _data[i]; ++i) {
				var zac = o.start.split(/\D/),
					kon = (o.end || o.start).split(/\D/);

				// dny pro validRange
				var den_zac = [
					zac[d.Rok],
					('0' + zac[d.Mes]).slice(-2), // mesic pro jistotu doplnim na delku min. 2
					'01' // den je vzdy 01 - povoluji zobrazit cele mesice
				].join('-');
				// cas 'do' je nevcetne, musim proto vypocitat prvni den nasledujiciho mesice
				// toto neni uplne dobre, pokud posledni udalost konci 31. 7. 23:59:59 - pak se
				// pro kalendar generuje datum do 1. 8. 00:00:00, jenze ja takto zbytecne prodlouzim
				// povolene konani do 1. 9. nevcetne (tj. do 31. 8.)
				var den_kon = [
					parseInt(kon[d.Rok], 10) + Math.floor(kon[d.Mes] / 12), // rok, prip. +1, je-li toto prosinec
					('0' + (1 + (kon[d.Mes] % 12))).slice(-2), // mesic zvetseny o 1
					'01', // den je opet vzdy 01
				].join('-');

				// zacatek a konec behem dne pouze pro udalosti netrvajici cely den
				if (!o.allDay && zac.length > 3) {
					minTime = Math.min(minTime, zac[d.Hod]);

					if (o.end) {
						// zname konec trvani udalosti
						// pokud udalost nekonci 13:00:00 (tj. ve skutecnosti 12:59:59), ale 13:37:42,
						// tak povolim konani az do 14:00 (to je to pripocteni 1)
						maxTime = Math.max(maxTime, (kon[d.Min] || kon[d.Sek] ? 1 : 0) + parseInt(kon[d.Hod], 10));
					} else {
						// konec je neznamy - kalendar zobrazi udalost na 2 hodiny
						maxTime = Math.max(maxTime, parseInt(zac[d.Hod], 10) + 2);
					}
				}

				if (!validRange.start || validRange.start > den_zac) {
					validRange.start = den_zac;
				}

				if (!validRange.end || validRange.end < den_kon) {
					validRange.end = den_kon;
				}
			}

			minTime = Math.max(0, minTime - 1) + ':00:00';
			maxTime = Math.min(24, maxTime + 1) + ':00:00';

			try {
				$('#kalendar').fullCalendar({
					locale: is.session.get('lang'),
					header: {
						left: 'title prev next today',
						right: 'month agendaWeek agendaDay listWeek'
					},
					eventTimeFormat: 'H:mm',
					eventLimit: true,
					events: _data,
					validRange: validRange,
					minTime: minTime,
					maxTime: maxTime,
					defaultView: posledni_typ_zobrazeni, // toto urcite existuje, server nikdy neposila nedefinovanou hodnotu
					eventRender: function(ev, el) {
						var $t = $(el).find('.fc-title');
						$t.html($t.text()); // titulek muze obsahovat HTML, napr. entitu &nbsp;, tu chci interpretovat
					},
					eventAfterAllRender: function(view) {
						// poznacim novy typ zobrazeni; i do preferenci
						is.Ajax.request('/auth/calendar/typ_zobrazeni', { t: view.type }).done(function() {
							posledni_typ_zobrazeni = view.type;
						});

						// po vykresleni vsech udalosti pridam zeleny tooltip
						// plati pouze pro zobrazeni mesice nebo tydne
						if (view.type === 'month' || view.type === 'agendaWeek') {

									popis = $(this).find('.fc-title').text();

					}).map(function(c) {
						return c.substr(3);
					});

						zobraz_udalosti(t);
					} else {

						skryj_udalosti(t);
					}

					// neni to jinak vychytrale reseni, ale na tuto nekritickou vec staci

						});
